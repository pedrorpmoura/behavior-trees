behavior : [
    selector : [
        sequence : [
            condition : $full_hp,
            action : $sub_hp
        ],
        action : $add_hp
    ]
]


%%
def full_hp(entity):
    return entity['hp'] == 100

def sub_hp(entity):
    entity['hp'] -= 10

def add_hp(entity):
    entity['hp'] += 10
